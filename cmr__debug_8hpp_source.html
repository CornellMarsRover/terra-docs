294template <typename U, typename T>
295auto narrow_cast(T&& t)
296{
297 using ValT = std::remove_reference<T>;
298 if constexpr (std::is_signed_v<ValT> && std::is_unsigned_v<U>) {
299 CMR_ASSERT_MSG(t >= 0, "Cannot narrow negative value");
300 } else if constexpr (std::is_unsigned_v<ValT> && std::is_signed_v<U>) {
301 CMR_ASSERT_MSG(static_cast<U>(std::forward<T>(t)) >= 0,
302 "Cannot narrow unsigned value");
303 }
304 const auto u = static_cast<U>(std::forward<T>(t));
305 CMR_ASSERT_MSG(static_cast<ValT>(u) == t, "Cannot narrow value");
306 return u;
307}
