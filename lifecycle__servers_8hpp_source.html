359 template <typename CallbackT, typename ErrorFuncT, typename ReturnT,
360 typename... ArgsT>
361 auto create_function_wrapper(CallbackT&& callback, ErrorFuncT&& error_cb,
362 std::unique_ptr<std::tuple<ArgsT...>>&&,
363 ReturnT error_val)
364 {
365 return [this, callback = std::forward<CallbackT>(callback),
366 error_cb = std::forward<ErrorFuncT>(error_cb),
367 error_val](ArgsT... args) {
368 if (m_active) {
369 // m_active shouldn't be false, but check this as a sanity check
370 // in case ROS copied the subscriber shared_ptr
371 try {
372 return callback(std::forward<ArgsT>(args)...);
373 } catch (const cmr_catch_t& e) {
374 RCLCPP_ERROR(m_node.get().get_logger(),
375 "An error occurred in a server callback: '%s'",
376 e.what());
377 if (++m_error_count > error_threshold) {
378 RCLCPP_ERROR(
379 m_node.get().get_logger(),
380 "More than %u uncaught exceptions occurred in "
381 "the same server callback. This happens when a "
382 "callback repeatedly errors, which should only "
383 "happen in extremely exceptional circumstances. "
384 "Exceptions should not be used for normal "
385 "control flow, and the restart behavior here "
386 "should not be relied upon for correct logic. Repeated "
387 "failures means there is a bug in your code.",
388 error_threshold);
389 }
390 error_cb();
391 }
392 }
393 using return_type_t = std::remove_reference_t<decltype(callback(
394 std::forward<ArgsT>(args)...))>;
395 if constexpr (!std::is_same_v<return_type_t, void>) {
396 return error_val;
397 } else {
398 // disable compiler warning of unused parameter for void functions
399 (void)error_val;
400 }
401 };
402 }
