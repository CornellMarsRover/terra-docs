19template <typename... Args>
20std::string string_format(const std::string& format, Args&&... args)
21{
22 // Source: https://stackoverflow.com/a/26221725/3212994 (explanation commented
23 // here too) the goal of this approach is to create a char buffer of the final
24 // string's size and use std::snprintf to write to it, before finally converting
25 // it to a std:string.
26
27 // first, we need to know what the final size actually is.
28 // this requires a call to std::snprintf, where we don't actually write the
29 // result anywhere but just use the size it returns. this also gives us a chance
30 // to do some error handling which is the advantage of this longer-winded
31 // approach over std::sprintf.
32 const int size_s =
33 std::snprintf(nullptr, 0, format.c_str(), std::forward<Args>(args)...) +
34 1; // add extra space for terminator '\0'
35 CMR_ASSERT_MSG(
36 size_s > 0,
37 "Invalid format or mismatched arguments passed to `string_format`");
38 const auto size = static_cast<size_t>(size_s);
39
40 // now that we know the size, and we know that the format is valid, we finally
41 // create our char buffer and use std::snprintf to write it
42 std::string buf(size, '\0'); // Use string for NRVO
43 std::snprintf(buf.data(), size, format.c_str(), std::forward<Args>(args)...);
44 buf.resize(size - 1);
45 // We don't want the terminator '\0' inside
46 return buf;
47}
